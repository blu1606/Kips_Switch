- **Seed Phrase:** 12/24 words from BIP39 list.
- **Credit Card:** 16 digits, Luhn check.

```typescript
const DANGER_PATTERNS = [
  {
    name: 'Solana Private Key',
    regex: /[1-9A-HJ-NP-Za-km-z]{87,88}/,
    suggestion: 'Private Keys should be encrypted.'
  },
  {
    name: 'Seed Phrase',
    regex: /^([a-z]{3,}\s){11,23}[a-z]{3,}$/i, // Basic check
    suggestion: 'Seed phrases must never be public.'
  }
];

function scanInput(text: string) {
  for (const pattern of DANGER_PATTERNS) {
    if (pattern.regex.test(text)) {
      return { detected: true, type: pattern.name };
    }
  }
  return { detected: false };
}
```
